Interface IMIOptions

Represents options for executing an MI request. See IMIRequest.

import { IMIOptions } from '@infor-up/m3-odin';

Example

const request = {
   program: 'CRS610MI',
   transaction: 'LstByNumber',
   excludeEmptyValues: true,
   maxReturnedRecords: 10
} as IMIRequest;
